This entry will include the widget built by this builder in the overlay at the entry's position.

To cause this builder to be called again, call markNeedsBuild on this overlay entry.

@override
WidgetBuilder get builder => (context) => Material(
      color: Colors.transparent,
      child: ValueListenableBuilder(
        valueListenable: _animation,
        builder: (context, value, child) {
          return Stack(
            fit: StackFit.expand,
            children: [
              GestureDetector(
                onTap: () {
                  _animation.value = 0.0;
                  Future.delayed(duration).then((value) => remove());
                },
                child: const ColoredBox(color: Colors.transparent),
              ),
              // Display the child widget
              this.child(context, value, this),
            ],
          );
        },
      ),
    );
