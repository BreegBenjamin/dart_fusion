static T of<T extends Object>(BuildContext context) {
  final widget = context.dependOnInheritedWidgetOfExactType<DProvider<T>>();
  if (widget == null) {
    throw FlutterError("No DProvider<$T> found in context");
  }
  return widget.value;
}
